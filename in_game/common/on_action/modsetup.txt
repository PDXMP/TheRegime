
on_game_start = {
    effect = {
        #catholics
		if = {
			limit = { NOT = { has_global_variable_list = pope_countries } }
			every_country = {
				limit = { has_reform = government_reform:papacy_reform }
				add_to_global_variable_list = { name = pope_countries target = this }
				#In case we need multiple papacies with antipopes and such
			}
		}
		every_country = {
			limit = {
				religion = religion:catholic
				not = {
					is_member_of_international_organization_of_type = { type = catholic_church }
					#they get added at the end
					this = c:ARA
				}
			}
			international_organization:catholic_church = {
				add_country_to_international_organization_no_update = prev
			}
		}
		international_organization:catholic_church = {
			#just do one update at the end
			add_country_to_international_organization = c:ARA
		}

		if = {
			limit = { current_year < 1338 }
			every_country = {
				limit = { is_ai = yes }
				random = {
					chance = 90
					add_cooldown = {
						type = ai_spam_protection
						years = 4	#4 years as it gets updated on four yearly pulse
					}
				}
			}
		}

		#shinto
		every_country = {
			limit = {
				religion = religion:shinto
				not = {
					is_member_of_international_organization_of_type = { type = shinto }
					#they get added at the end
					this = c:STC
				}
			}
			international_organization:shinto = {
				add_country_to_international_organization_no_update = prev
			}
		}
		international_organization:shinto = {
			#just do one update at the end
			add_country_to_international_organization = c:STC
		}
		
		# HRE
		international_organization:hre.leader_country ?= {
			set_variable = { name = hre_current_emperor_flg value = 1 }
		}

		#Adding gods to pagans
		every_country = {
			limit = {
				is_country_religion_pagan = yes
			}
			save_scope_as = current_country
			religion = {
				#Add every god of their religion to the country
				every_god_in_religion = {
					limit = {
						#Add any restriction for the addition of gods
						OR = {
							NOT = { this = god:huitzilopochtli_god }
							AND = {
								this = god:huitzilopochtli_god
								scope:current_country = location:tenochtitlan.owner
							}
						}
					}
					scope:current_country = {
						add_god = prev
					}
				}
			}	
		}		

		c:CHI = {
			china_set_up = yes
		}
		c:MAJ = {
			majapahit_set_up = yes
		}
		c:JAP = {
			japan_set_up = yes
		}
    c:BOH = {
        revoke_estate_privilege = estate_privilege:auxilium_et_consilium
        grant_estate_privilege = estate_privilege:monetary_fiefs
        grant_estate_privilege = estate_privilege:noble_leaders_only
        grant_estate_privilege = estate_privilege:primacy_of_nobility
        cancel_subject = c:WRC
        cancel_subject = c:ZIE
        cancel_subject = c:ZAG
        cancel_subject = c:STZ
        cancel_subject = c:SIZ
        cancel_subject = c:SCI
        cancel_subject = c:RCB
        cancel_subject = c:PDK
        cancel_subject = c:OSW
        cancel_subject = c:OPL
        cancel_subject = c:OPV
        cancel_subject = c:OEL
        cancel_subject = c:NMD
        cancel_subject = c:NMS
        cancel_subject = c:LGC
        cancel_subject = c:GLO
        cancel_subject = c:TES
        cancel_subject = c:BYM
        cancel_subject = c:BRZ
        cancel_subject = c:MVA
        create_union = c:MVA
    }
    c:FRA = {
            trigger_event_non_silently = setup_mod.1
        }
    }
}
